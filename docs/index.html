use-react-screenshot

hook which allows to create screenshots

Install

npm install --save use-react-screenshot

Usage

import React, { Component } from 'react'

import { useMyHook } from 'use-react-screenshot'

const Example = () => {
  const example = useMyHook()
  return (
    <div>{example}</div>
  )
}
